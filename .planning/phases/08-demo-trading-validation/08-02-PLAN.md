Create `scripts/run_bot.ps1` -- a PowerShell script that wraps `python main.py` with auto-restart:

```powershell
# GoldBot 2 -- Process Restart Wrapper
# Automatically restarts the bot after a crash with cooldown.
# Usage: powershell -ExecutionPolicy Bypass -File scripts/run_bot.ps1

param(
    [int]$MaxRestarts = 50,
    [int]$CooldownSeconds = 30,
    [switch]$Preflight
)

$ErrorActionPreference = "Stop"
$scriptDir = Split-Path -Parent $MyInvocation.MyCommand.Path
$projectDir = Split-Path -Parent $scriptDir

Push-Location $projectDir

# Optional pre-flight check
if ($Preflight) {
    Write-Host "Running pre-flight checks..." -ForegroundColor Cyan
    python scripts/preflight_check.py
    if ($LASTEXITCODE -ne 0) {
        Write-Host "Pre-flight checks FAILED. Fix issues before starting." -ForegroundColor Red
        Pop-Location
        exit 1
    }
    Write-Host "Pre-flight checks PASSED." -ForegroundColor Green
    Write-Host ""
}

$restartCount = 0
$logFile = "logs/restart_log.txt"

while ($restartCount -lt $MaxRestarts) {
    $attempt = $restartCount + 1
    $timestamp = Get-Date -Format "yyyy-MM-dd HH:mm:ss"
    $msg = "$timestamp | Starting GoldBot (attempt $attempt/$MaxRestarts)..."
    Write-Host $msg -ForegroundColor Green
    Add-Content -Path $logFile -Value $msg

    python main.py
    $exitCode = $LASTEXITCODE
    $restartCount++

    $timestamp = Get-Date -Format "yyyy-MM-dd HH:mm:ss"

    if ($exitCode -eq 0) {
        $msg = "$timestamp | Bot exited cleanly (exit code 0). Stopping."
        Write-Host $msg -ForegroundColor Green
        Add-Content -Path $logFile -Value $msg
        break
    }

    $msg = "$timestamp | Bot crashed (exit code $exitCode). Restart $restartCount/$MaxRestarts. Cooldown ${CooldownSeconds}s..."
    Write-Host $msg -ForegroundColor Yellow
    Add-Content -Path $logFile -Value $msg

    Start-Sleep -Seconds $CooldownSeconds
}

if ($restartCount -ge $MaxRestarts) {
    $msg = "$(Get-Date -Format 'yyyy-MM-dd HH:mm:ss') | Max restarts ($MaxRestarts) reached. Manual intervention required."
    Write-Host $msg -ForegroundColor Red
    Add-Content -Path $logFile -Value $msg
}

Pop-Location
```

Key features:
- `-Preflight` switch runs preflight_check.py before starting
- `-MaxRestarts 50` prevents infinite restart loops
- `-CooldownSeconds 30` prevents rapid restarts
- Logs all restart events to `logs/restart_log.txt`
- Clean exit (code 0) stops the wrapper
- Non-zero exit triggers restart with cooldown
