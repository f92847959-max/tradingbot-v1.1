     b. **Daily stats aggregation** (runs at midnight UTC, inside the `if _last_cleanup_date != today and now_utc.hour == 0:` block):
     ```python
     # Aggregate yesterday's stats
     try:
         yesterday = today - timedelta(days=1)
         async with get_session() as session:
             trade_repo = TradeRepository(session)
             stats_repo = StatsRepository(session)
             # Get yesterday's closed trades
             from sqlalchemy import select, and_, func as sqlfunc
             from database.models import Trade
             stmt = (
                 select(Trade)
                 .where(and_(
                     sqlfunc.date(Trade.closed_at) == yesterday,
                     Trade.status == "CLOSED",
                 ))
             )
             result = await session.execute(stmt)
             trades = result.scalars().all()
             if trades:
                 winners = [t for t in trades if float(t.net_pnl or 0) > 0]
                 losers = [t for t in trades if float(t.net_pnl or 0) < 0]
                 gross_profit = sum(float(t.net_pnl or 0) for t in winners)
                 gross_loss = sum(float(t.net_pnl or 0) for t in losers)
                 net = gross_profit + gross_loss
                 await stats_repo.upsert({
                     "date": yesterday,
                     "trades_total": len(trades),
                     "trades_won": len(winners),
                     "trades_lost": len(losers),
                     "win_rate": len(winners) / len(trades) if trades else 0,
                     "gross_profit": round(gross_profit, 2),
                     "gross_loss": round(gross_loss, 2),
                     "net_pnl": round(net, 2),
                     "profit_factor": round(abs(gross_profit / gross_loss), 4) if gross_loss else None,
                 })
                 logger.info("Daily stats aggregated for %s: %d trades, net=%.2f", yesterday, len(trades), net)
     except Exception as e:
         logger.error("Daily stats aggregation failed: %s", e)
     ```
