    - Class PortfolioHeatManager:
      __init__(self, max_heat_pct: float = 5.0):
        self.max_heat_pct = max_heat_pct
        self._open_risk_total: float = 0.0  # Sum of all open position risk amounts

      add_position(self, risk_amount: float, account_balance: float) -> None:
        self._open_risk_total += risk_amount
        heat = self.get_heat(account_balance)
        logger.info("Position added: risk=%.2f, heat=%.2f%% / %.2f%%", risk_amount, heat, self.max_heat_pct)

      remove_position(self, risk_amount: float, account_balance: float) -> None:
        self._open_risk_total = max(0.0, self._open_risk_total - risk_amount)
        logger.info("Position removed: risk=%.2f, heat=%.2f%%", risk_amount, self.get_heat(account_balance))

      get_heat(self, account_balance: float) -> float:
        if account_balance <= 0: return 0.0
        return (self._open_risk_total / account_balance) * 100.0

      can_add_position(self, risk_amount: float, account_balance: float) -> bool:
        if account_balance <= 0: return False
        projected = ((self._open_risk_total + risk_amount) / account_balance) * 100.0
        return projected <= self.max_heat_pct

      get_remaining_heat(self, account_balance: float) -> float:
        """Returns dollar amount of risk capacity remaining."""
        max_risk = account_balance * (self.max_heat_pct / 100.0)
        return max(0.0, max_risk - self._open_risk_total)

      reset(self) -> None:
        self._open_risk_total = 0.0

    Create risk/equity_curve_filter.py:
    - Class EquityCurveFilter:
      __init__(self, ema_period: int = 20, enabled: bool = True):
        self.ema_period = ema_period
        self.enabled = enabled
        self._equity_history: list[float] = []
        self._ema: float = 0.0

      update(self, equity: float) -> bool:
        """Add equity point, recalculate EMA, return is_trading_allowed."""
        self._equity_history.append(equity)
        if len(self._equity_history) >= self.ema_period:
            # EMA calculation: multiplier = 2 / (period + 1)
            k = 2.0 / (self.ema_period + 1)
            if self._ema == 0.0:
                # Seed EMA with SMA of first ema_period points
                self._ema = sum(self._equity_history[-self.ema_period:]) / self.ema_period
            else:
                self._ema = equity * k + self._ema * (1 - k)
        return self.is_trading_allowed()

      is_trading_allowed(self) -> bool:
        if not self.enabled: return True
        if len(self._equity_history) < self.ema_period: return True  # Insufficient data
        current_equity = self._equity_history[-1]
        allowed = current_equity >= self._ema
        if not allowed:
            logger.warning("Equity curve filter: equity=%.2f < EMA=%.2f -- trading restricted", current_equity, self._ema)
        return allowed

      get_ema(self) -> float:
        return self._ema

      get_equity_vs_ema(self) -> str:
        if len(self._equity_history) < self.ema_period: return "insufficient_data"
        return "above" if self._equity_history[-1] >= self._ema else "below"

      reset(self) -> None:
        self._equity_history.clear()
        self._ema = 0.0
