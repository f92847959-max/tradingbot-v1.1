    ```python
    """Correlation calculations for Phase 12 (CORR-02, CORR-03).

    Pure-function kernel: DataFrame -> CorrelationSnapshot. No I/O.
    """
    from __future__ import annotations

    import numpy as np
    import pandas as pd

    from correlation.snapshot import CorrelationSnapshot

    WINDOWS = [20, 60, 120]
    ASSETS_VS_GOLD = ["dxy", "us10y", "silver", "vix", "sp500"]


    def rolling_corr(a: pd.Series, b: pd.Series, window: int) -> float:
        """Pearson correlation of the last `window` aligned observations."""
        if len(a) < window or len(b) < window:
            return 0.0
        aligned = pd.concat([a, b], axis=1).dropna()
        if len(aligned) < window:
            return 0.0
        val = aligned.iloc[:, 0].rolling(window).corr(aligned.iloc[:, 1]).iloc[-1]
        if pd.isna(val):
            return 0.0
        return float(val)


    def divergence_score(a: pd.Series, b: pd.Series, lookback: int = 20) -> float:
        """Fraction of last `lookback` bars where returns have OPPOSITE signs.

        Range [0.0, 1.0]. 0.0 = returns move together; 1.0 = always opposite.
        """
        if len(a) < lookback + 1 or len(b) < lookback + 1:
            return 0.0
        ra = a.pct_change().iloc[-lookback:]
        rb = b.pct_change().iloc[-lookback:]
        if ra.empty or rb.empty:
            return 0.0
        # Sign agreement: 1.0 when same sign, 0.0 when opposite sign
        same_sign = (np.sign(ra.values) == np.sign(rb.values)).astype(float)
        # Divergence = fraction of OPPOSITE signs
        return float(1.0 - same_sign.mean())


    def lead_lag(a: pd.Series, b: pd.Series, max_lag: int = 10) -> float:
        """Cross-correlation peak lag, normalized to [-1.0, 1.0].

        Sign convention (RESEARCH Pitfall 5):
          - negative: `a` leads `b` (gold leads other asset)
          - positive: `b` leads `a` (other asset leads gold)
        Returns `-best_lag / max_lag`.
        """
        if len(a) < 2 * max_lag + 1 or len(b) < 2 * max_lag + 1:
            return 0.0
        a_norm = (a - a.mean()) / (a.std() + 1e-12)
        b_norm = (b - b.mean()) / (b.std() + 1e-12)
        xcorr = np.correlate(a_norm.values, b_norm.values, mode="full")
        # Center of xcorr is index len(a)-1; lags range over [-(N-1), +(N-1)]
        center = len(a) - 1
        window = xcorr[center - max_lag : center + max_lag + 1]
        best_lag = int(np.argmax(np.abs(window))) - max_lag
        return float(-best_lag / max_lag)


    def regime(current_corr: float, recent_corrs: pd.Series) -> float:
        """Classify current correlation vs. historical distribution.

        Returns:
          1.0  = breakdown (z > 2.0, current far weaker/stronger than history)
          -1.0 = inversion (z < -2.0 or sign flip)
          0.0  = normal
        """
        if len(recent_corrs) < 10:
            return 0.0
        mean = recent_corrs.mean()
        std = recent_corrs.std()
        if std < 1e-9:
            return 0.0
        z = (current_corr - mean) / std
        if z > 2.0:
            return 1.0
        if z < -2.0:
            return -1.0
        return 0.0


    def compute_snapshot(closes: pd.DataFrame) -> CorrelationSnapshot:
        """Transform DataFrame(dxy/us10y/silver/vix/sp500/gold) -> CorrelationSnapshot.

        Graceful degradation: missing columns or insufficient rows yield 0.0 fields.
        """
        if closes is None or closes.empty or "gold" not in closes.columns:
            return CorrelationSnapshot()

        gold = closes["gold"]
        fields: dict[str, float] = {}

        # Rolling correlations across all windows x assets (15 fields)
        for asset in ASSETS_VS_GOLD:
            if asset not in closes.columns:
                for w in WINDOWS:
                    fields[f"corr_{asset}_{w}"] = 0.0
                continue
            for w in WINDOWS:
                fields[f"corr_{asset}_{w}"] = rolling_corr(gold, closes[asset], w)

        # Divergence scores (2 fields) — DXY + US10Y only
        for asset in ["dxy", "us10y"]:
            if asset in closes.columns:
                fields[f"divergence_{asset}"] = divergence_score(gold, closes[asset])
            else:
                fields[f"divergence_{asset}"] = 0.0

        # Regime — use DXY 60-bar correlation as canonical signal
        if "dxy" in closes.columns and len(closes) >= 60:
            recent = gold.rolling(60).corr(closes["dxy"]).dropna().iloc[-30:]
            if not recent.empty:
                fields["corr_regime"] = regime(fields["corr_dxy_60"], recent)
            else:
                fields["corr_regime"] = 0.0
        else:
            fields["corr_regime"] = 0.0

        # Lead-lag (2 fields) — silver + dxy
        for asset in ["silver", "dxy"]:
            if asset in closes.columns:
                fields[f"lead_lag_{asset}"] = lead_lag(gold, closes[asset])
            else:
                fields[f"lead_lag_{asset}"] = 0.0

        return CorrelationSnapshot(**fields)
    ```
