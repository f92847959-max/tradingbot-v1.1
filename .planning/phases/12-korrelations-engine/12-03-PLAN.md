    **Step 2a — Create `ai_engine/features/correlation_features.py`:**
    ```python
    """Correlation feature group (CORR-04).

    Materialises a CorrelationSnapshot into 20 DataFrame columns. Every row in
    the DataFrame receives the same scalar value per field — the snapshot is a
    point-in-time inter-market measurement, broadcast across all intraday gold
    candles for this feature cycle.
    """
    from __future__ import annotations

    import logging
    from dataclasses import asdict, fields
    from typing import List, Optional

    import pandas as pd

    from correlation.snapshot import CorrelationSnapshot

    logger = logging.getLogger(__name__)


    # Canonical feature order — derived from CorrelationSnapshot fields at import time.
    _FEATURE_NAMES: List[str] = [f.name for f in fields(CorrelationSnapshot)]


    class CorrelationFeatures:
        """Broadcast CorrelationSnapshot scalars into 20 DataFrame columns."""

        def __init__(self) -> None:
            self._feature_names: List[str] = list(_FEATURE_NAMES)
            logger.debug(
                "CorrelationFeatures initialised with %d features", len(self._feature_names),
            )

        def get_feature_names(self) -> List[str]:
            return list(self._feature_names)

        def calculate(
            self,
            df: pd.DataFrame,
            snapshot: Optional[CorrelationSnapshot] = None,
        ) -> pd.DataFrame:
            """Add 20 correlation columns to `df`.

            When `snapshot` is None all columns are filled with 0.0
            (graceful degradation — backtest warmup, fetcher failure, etc).
            """
            if snapshot is None:
                values = {name: 0.0 for name in self._feature_names}
            else:
                values = asdict(snapshot)

            for name in self._feature_names:
                df[name] = float(values.get(name, 0.0))
            return df
    ```
