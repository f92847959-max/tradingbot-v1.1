    """Sentiment Features -- 6 ML features from news sentiment pipeline.

    Mirrors the FEATURE_NAMES + calculate(df) pattern of TechnicalFeatures.
    Queries SentimentAggregator live (bypasses FeatureEngineer cache) because
    sentiment changes independently of candle timestamps (see CONTEXT.md D-13,
    RESEARCH Pitfall 3).
    """

    from __future__ import annotations

    import logging
    from typing import List, Optional, TYPE_CHECKING

    import numpy as np
    import pandas as pd

    if TYPE_CHECKING:
        from sentiment.sentiment_aggregator import SentimentAggregator

    logger = logging.getLogger(__name__)


    class SentimentFeatures:
        """Generates 6 sentiment-based ML features.

        All features are in [-1.0, +1.0] except news_count_1h which is a
        non-negative integer count. Feature values are identical across
        all rows in the returned DataFrame (point-in-time snapshot at the
        latest timestamp) -- matching how existing features treat forward-
        looking quantities.
        """

        FEATURE_NAMES: List[str] = [
            "sent_1h",          # EWM-averaged sentiment last 1h (-1..+1)
            "sent_4h",          # EWM-averaged sentiment last 4h
            "sent_24h",         # EWM-averaged sentiment last 24h
            "sent_momentum",    # sent_1h - sent_4h (rate of change)
            "sent_divergence",  # sent_1h - price_direction (-1..+1; CONTEXT D-11)
            "news_count_1h",    # integer count of articles in last 1h
        ]

        def __init__(self, aggregator: Optional["SentimentAggregator"] = None) -> None:
            """aggregator=None => all sentiment features = 0.0 (graceful)."""
            self._aggregator = aggregator

        def get_feature_names(self) -> List[str]:
            return self.FEATURE_NAMES.copy()

        def calculate(self, df: pd.DataFrame) -> pd.DataFrame:
            """Add 6 sentiment columns to df.

            Args:
                df: DataFrame with at least a `close` column and a
                    DatetimeIndex (or a `timestamp` column). Other rows
                    are allowed but only the latest timestamp is used
                    for the point-in-time lookup.

            Returns:
                Copy of df with 6 new columns. All values identical
                across rows (snapshot).
            """
            df = df.copy()

            # Graceful fallback when sentiment disabled (CONTEXT D-13)
            if self._aggregator is None:
                for col in self.FEATURE_NAMES:
                    df[col] = 0.0
                return df

            # Determine the "current" timestamp for point-in-time lookup
            if isinstance(df.index, pd.DatetimeIndex) and len(df.index):
                latest_ts = df.index[-1].to_pydatetime()
                if latest_ts.tzinfo is None:
                    latest_ts = latest_ts.replace(tzinfo=__import__("datetime").timezone.utc)
            elif "timestamp" in df.columns and len(df):
                ts = pd.to_datetime(df["timestamp"].iloc[-1], utc=True)
                latest_ts = ts.to_pydatetime()
            else:
                # No timestamp information -- return zeros
                for col in self.FEATURE_NAMES:
                    df[col] = 0.0
                return df

            try:
                # window_records=None -> aggregator falls back to its own repo
                features = self._aggregator.get_features_at(latest_ts, window_records=None)
            except Exception as exc:  # defensive: never crash feature pipeline
                logger.warning("SentimentAggregator failed, zero-fallback: %s", exc)
                features = {k: 0.0 for k in self.FEATURE_NAMES if k != "sent_divergence"}

            # Compute price direction for divergence (CONTEXT D-11)
            price_direction = 0.0
            if "close" in df.columns and len(df) >= 2:
                prev_close = float(df["close"].iloc[-2])
                curr_close = float(df["close"].iloc[-1])
                if curr_close > prev_close:
                    price_direction = 1.0
                elif curr_close < prev_close:
                    price_direction = -1.0

            sent_1h = float(features.get("sent_1h", 0.0))
            sent_4h = float(features.get("sent_4h", 0.0))
            sent_24h = float(features.get("sent_24h", 0.0))
            sent_momentum = float(features.get("sent_momentum", sent_1h - sent_4h))
            news_count_1h = float(features.get("news_count_1h", 0.0))
            sent_divergence = max(-1.0, min(1.0, sent_1h - price_direction))

            df["sent_1h"] = sent_1h
            df["sent_4h"] = sent_4h
            df["sent_24h"] = sent_24h
            df["sent_momentum"] = sent_momentum
            df["sent_divergence"] = sent_divergence
            df["news_count_1h"] = news_count_1h

            return df
    ```

    """Writes a German-prose Markdown seed for MiroFish from recent sentiment.

    Follows the style of mirofish_seeds/gold_market_overview.md (deutsche Prosa,
    # Titel, ## Abschnitte, keine rohen Daten-Dumps).

    from __future__ import annotations

    import logging
    from datetime import datetime, timedelta, timezone
    from pathlib import Path
    from typing import Iterable, Sequence

    from sentiment.sentiment_aggregator import SentimentAggregator

    logger = logging.getLogger(__name__)


    def _label(score: float) -> str:
        if score >= 0.2:
            return "positiv"
        if score <= -0.2:
            return "negativ"
        return "neutral"


    def write_sentiment_seed(
        records: Sequence[dict],
        path: Path | str,
        now: datetime | None = None,
        halflife_minutes: float = 30.0,
    ) -> None:
        """Render MiroFish seed from sentiment records.

        Args:
            records: iterable of dicts with keys `published_at`, `sentiment_score`,
                `source_weight`, `headline`, `source`.
            path: destination file.
            now: reference timestamp (defaults to utcnow()).
            halflife_minutes: passed to SentimentAggregator for aggregation.
        """
        now = now or datetime.now(timezone.utc)
        agg = SentimentAggregator(halflife_minutes=halflife_minutes)
        features = agg.get_features_at(now, window_records=list(records))

        sent_1h = features["sent_1h"]
        sent_4h = features["sent_4h"]
        sent_24h = features["sent_24h"]
        count_1h = int(features["news_count_1h"])

        # Pick top 3 most recent articles (news impact recency)
        sorted_records = sorted(
            records,
            key=lambda r: r["published_at"] if isinstance(r, dict) else r.published_at,
            reverse=True,
        )
        top_headlines = sorted_records[:3]

        lines: list[str] = []
        lines.append("# Sentiment-Ueberblick Gold (XAUUSD)")
        lines.append("")
        lines.append(
            f"Stand: {now.strftime('%Y-%m-%d %H:%M UTC')}. "
            f"Dieser Seed wird stuendlich aus Nachrichten-Sentiment aktualisiert."
        )
        lines.append("")
        lines.append("## Aggregierte Stimmung")
        lines.append("")
        lines.append(
            f"- Letzte Stunde: Score {sent_1h:+.2f} ({_label(sent_1h)}), "
            f"{count_1h} Meldungen."
        )
        lines.append(f"- Letzte 4 Stunden: Score {sent_4h:+.2f} ({_label(sent_4h)}).")
        lines.append(f"- Letzte 24 Stunden: Score {sent_24h:+.2f} ({_label(sent_24h)}).")
        lines.append("")
        lines.append("## Top-Schlagzeilen")
        lines.append("")
        if not top_headlines:
            lines.append("Aktuell keine gold-relevanten Schlagzeilen im 24h-Fenster.")
        else:
            for r in top_headlines:
                headline = r["headline"] if isinstance(r, dict) else r.headline
                source = r.get("source", "?") if isinstance(r, dict) else getattr(r, "source", "?")
                lines.append(f"- [{source}] {headline}")
        lines.append("")
        lines.append("## Kontext")
        lines.append("")
        lines.append(
            "Die Werte dienen als zusaetzlicher Kontext fuer die MiroFish-"
            "Agenten. Negative Scores deuten auf bearishe News, positive auf "
            "bullishe. Bei stark divergierendem 1h vs 4h Score liegt ein "
            "kurzfristiger Stimmungswechsel vor."
        )
        lines.append("")

        Path(path).parent.mkdir(parents=True, exist_ok=True)
        Path(path).write_text("\n".join(lines), encoding="utf-8")
        logger.info("Sentiment seed written: %s", path)

    """Sentiment Service facade -- orchestrates FeedFetcher + Analyzer + Aggregator + Repo.

    Wired into trading/lifecycle.py (start/stop) when settings.sentiment_enabled.
    """
    from __future__ import annotations

    import asyncio
    import logging
    from datetime import datetime, timedelta, timezone
    from pathlib import Path
    from typing import Any, Callable, Optional

    from apscheduler.schedulers.asyncio import AsyncIOScheduler

    from sentiment.news_fetcher import FeedFetcher, GOLD_FEEDS
    from sentiment.sentiment_analyzer import SentimentAnalyzer
    from sentiment.sentiment_aggregator import SentimentAggregator
    from sentiment.sentiment_repository import SentimentRepository
    from sentiment.seed_writer import write_sentiment_seed

    logger = logging.getLogger(__name__)


    class SentimentService:
        """Owns the APScheduler and all sentiment sub-modules."""

        def __init__(
            self,
            settings: Any,
            session_factory: Callable[[], Any],  # async context manager returning AsyncSession
            seed_path: str | Path = "mirofish_seeds/news_sentiment.md",
        ) -> None:
            self._settings = settings
            self._session_factory = session_factory
            self._seed_path = Path(seed_path)

            self._fetcher = FeedFetcher(
                source_weights=dict(settings.sentiment_source_weights),
                min_keywords=settings.sentiment_min_keywords,
            )
            self._analyzer = SentimentAnalyzer(
                model=settings.sentiment_model,
                finbert_cache_path=settings.sentiment_finbert_cache_path or "",
            )
            self._aggregator = SentimentAggregator(
                repository=None,
                halflife_minutes=float(settings.sentiment_halflife_minutes),
            )
            self._scheduler: Optional[AsyncIOScheduler] = None

        @property
        def aggregator(self) -> SentimentAggregator:
            return self._aggregator

        async def start(self) -> None:
            if self._scheduler is not None:
                return
            self._scheduler = AsyncIOScheduler(timezone="UTC")
            self._scheduler.add_job(
                self._poll_and_store,
                "interval",
                seconds=int(self._settings.sentiment_poll_interval_seconds),
                id="sentiment_poll",
                max_instances=1,
                coalesce=True,
            )
            self._scheduler.add_job(
                self._refresh_seed,
                "interval",
                hours=int(self._settings.sentiment_seed_update_hours),
                id="sentiment_seed",
                max_instances=1,
                coalesce=True,
            )
            self._scheduler.start()
            logger.info(
                "SentimentService started (poll=%ss, seed=%dh)",
                self._settings.sentiment_poll_interval_seconds,
                self._settings.sentiment_seed_update_hours,
            )

        async def stop(self) -> None:
            if self._scheduler is None:
                return
            try:
                self._scheduler.shutdown(wait=False)
            except Exception as exc:
                logger.warning("SentimentService shutdown error: %s", exc)
            finally:
                self._scheduler = None
                logger.info("SentimentService stopped")

        async def _poll_and_store(self) -> None:
            """Poll all feeds, score articles, store in DB. Runs every poll_interval."""
            all_articles = []
            for source, url in GOLD_FEEDS.items():
                try:
                    raw = await self._fetcher.poll_feed(source, url)
                    for art in raw:
                        art["sentiment_score"] = self._analyzer.score(
                            f"{art.get('headline', '')} {art.get('summary', '')}"
                        )
                        art["model_used"] = self._settings.sentiment_model
                        all_articles.append(art)
                except Exception as exc:  # defensive -- don't let one feed kill the loop
                    logger.warning("Feed poll failed for %s: %s", source, exc)

            if not all_articles:
                return

            try:
                async with self._session_factory() as session:
                    repo = SentimentRepository(session)
                    inserted = await repo.save_batch(all_articles)
                    logger.info("Sentiment poll: %d new articles", inserted)
                    # Best-effort retention prune
                    pruned = await repo.prune_older_than(
                        days=int(self._settings.sentiment_retention_days)
                    )
                    if pruned:
                        logger.debug("Pruned %d old sentiment rows", pruned)
            except Exception as exc:
                logger.error("Sentiment DB save failed: %s", exc)

        async def _refresh_seed(self) -> None:
            """Rebuild mirofish_seeds/news_sentiment.md from last 24h records."""
            now = datetime.now(timezone.utc)
            try:
                async with self._session_factory() as session:
                    repo = SentimentRepository(session)
                    records = await repo.get_records(since=now - timedelta(hours=24), until=now)
                # Convert ORM rows -> dicts for aggregator
                records_as_dicts = [
                    {
                        "published_at": r.published_at,
                        "sentiment_score": float(r.sentiment_score),
                        "source_weight": float(r.source_weight),
                        "headline": r.headline,
                        "source": r.source,
                    }
                    for r in records
                ]
                write_sentiment_seed(
                    records_as_dicts,
                    path=self._seed_path,
                    now=now,
                    halflife_minutes=float(self._settings.sentiment_halflife_minutes),
                )
            except Exception as exc:
                logger.error("Sentiment seed refresh failed: %s", exc)
    ```
