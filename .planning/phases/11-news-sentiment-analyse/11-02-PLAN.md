    """VADER / FinBERT scorer returning compound score in [-1.0, +1.0]."""
    from __future__ import annotations
    import logging
    import os
    import re
    from typing import Literal, Optional

    logger = logging.getLogger(__name__)
    _HTML_TAG_RE = re.compile(r"<[^>]+>")


    class SentimentAnalyzer:
        def __init__(
            self,
            model: Literal["vader", "finbert"] = "vader",
            finbert_cache_path: str = "",
        ) -> None:
            self._model_type = model
            if model == "vader":
                from vaderSentiment.vaderSentiment import SentimentIntensityAnalyzer
                self._vader = SentimentIntensityAnalyzer()
                self._pipe = None
            elif model == "finbert":
                if finbert_cache_path:
                    os.environ["TRANSFORMERS_CACHE"] = finbert_cache_path
                from transformers import pipeline
                self._pipe = pipeline(
                    "text-classification",
                    model="ProsusAI/finbert",
                    top_k=None,
                )
                self._vader = None
            else:
                raise ValueError(f"Unknown sentiment model: {model}")

        def score(self, text: str) -> float:
            if not text:
                return 0.0
            cleaned = _HTML_TAG_RE.sub(" ", text)
            cleaned = cleaned.strip()
            if not cleaned:
                return 0.0
            if self._model_type == "vader":
                s = self._vader.polarity_scores(cleaned)["compound"]
            else:
                results = self._pipe(cleaned[:512])[0]
                scores = {r["label"].lower(): r["score"] for r in results}
                s = scores.get("positive", 0.0) - scores.get("negative", 0.0)
            # Clamp defensive (VADER already in range, FinBERT math is in [-1,1])
            return max(-1.0, min(1.0, float(s)))
    ```

    """Time-decayed aggregation of sentiment scores into rolling windows."""
    from __future__ import annotations
    import logging
    from dataclasses import dataclass
    from datetime import datetime, timedelta, timezone
    from typing import Iterable, Optional

    import pandas as pd

    logger = logging.getLogger(__name__)


    @dataclass
    class SentimentRecord:
        published_at: datetime
        sentiment_score: float
        source_weight: float = 1.0


    class SentimentAggregator:
        """Compute EWM-weighted sentiment across 1h/4h/24h windows.

        Pure-logic: accepts a list of records and a reference timestamp,
        returns a dict of sent_* features. No I/O, no async. Repository
        feeds records from the DB in Plan 11-03.
        """

        FEATURE_KEYS = ("sent_1h", "sent_4h", "sent_24h", "sent_momentum", "news_count_1h")

        def __init__(
            self,
            repository: Optional["SentimentRepository"] = None,  # Optional for pure-logic tests
            halflife_minutes: float = 30.0,
        ) -> None:
            self._repo = repository
            self._halflife_minutes = halflife_minutes

        def compute_window(
            self,
            records: Iterable[dict | SentimentRecord],
            now: datetime,
            window_hours: float,
        ) -> float:
            """Single-window EWM. Returns 0.0 if empty."""
            cutoff = now - timedelta(hours=window_hours)
            rows = []
            for r in records:
                ts = r.published_at if isinstance(r, SentimentRecord) else r["published_at"]
                if ts < cutoff or ts > now:
                    continue
                score = r.sentiment_score if isinstance(r, SentimentRecord) else r["sentiment_score"]
                weight = r.source_weight if isinstance(r, SentimentRecord) else r.get("source_weight", 1.0)
                rows.append({"published_at": ts, "weighted": float(score) * float(weight)})
            if not rows:
                return 0.0
            df = pd.DataFrame(rows).sort_values("published_at")
            df["published_at"] = pd.to_datetime(df["published_at"], utc=True)
            ewm = df["weighted"].ewm(
                halflife=f"{self._halflife_minutes}min",
                times=df["published_at"],
            ).mean()
            val = float(ewm.iloc[-1])
            return max(-1.0, min(1.0, val))

        def count_window(
            self,
            records: Iterable[dict | SentimentRecord],
            now: datetime,
            window_hours: float,
        ) -> int:
            cutoff = now - timedelta(hours=window_hours)
            count = 0
            for r in records:
                ts = r.published_at if isinstance(r, SentimentRecord) else r["published_at"]
                if cutoff <= ts <= now:
                    count += 1
            return count

        def get_features_at(
            self,
            now: datetime,
            window_records: Iterable[dict | SentimentRecord] | None = None,
        ) -> dict[str, float]:
            """Return dict with 5 aggregated features (sent_divergence added in SentimentFeatures).

            If window_records is None and self._repo is set, fetches from repository
            in the caller's event loop (synchronous context). For async contexts,
            caller pre-fetches and passes window_records explicitly.
            """
            records = list(window_records) if window_records is not None else []
            if not records and self._repo is not None:
                # Synchronous fallback: repository expected to handle own awaiting
                records = self._repo.get_records_sync(now - timedelta(hours=24), now)
            sent_1h = self.compute_window(records, now, 1.0)
            sent_4h = self.compute_window(records, now, 4.0)
            sent_24h = self.compute_window(records, now, 24.0)
            return {
                "sent_1h": sent_1h,
                "sent_4h": sent_4h,
                "sent_24h": sent_24h,
                "sent_momentum": sent_1h - sent_4h,
                "news_count_1h": float(self.count_window(records, now, 1.0)),
            }

        async def get_features_at_async(
            self,
            now: datetime,
        ) -> dict[str, float]:
            """Async variant that pulls from repository."""
            if self._repo is None:
                return {k: 0.0 for k in self.FEATURE_KEYS}
            since = now - timedelta(hours=24)
            records = await self._repo.get_records(since=since, until=now)
            return self.get_features_at(now, window_records=records)
    ```

    ```python
    """Async SQLAlchemy repository for news_sentiment table."""
    from __future__ import annotations
    import logging
    from datetime import datetime, timedelta, timezone
    from typing import Iterable, Optional

    from sqlalchemy import delete, select
    from sqlalchemy.dialects.sqlite import insert as sqlite_insert
    from sqlalchemy.dialects.postgresql import insert as postgres_insert
    from sqlalchemy.exc import IntegrityError
    from sqlalchemy.ext.asyncio import AsyncSession

    from database.models import NewsSentiment

    logger = logging.getLogger(__name__)


    class SentimentRepository:
        def __init__(self, session: AsyncSession) -> None:
            self._session = session

        async def save_batch(self, articles: Iterable[dict]) -> int:
            """Insert articles; skip duplicates via ON CONFLICT. Returns inserted count."""
            articles = list(articles)
            if not articles:
                return 0
            # Dialect-aware insert with on_conflict_do_nothing on entry_id
            dialect = self._session.bind.dialect.name if self._session.bind else "sqlite"
            if dialect == "postgresql":
                stmt = postgres_insert(NewsSentiment).values(articles).on_conflict_do_nothing(
                    index_elements=["entry_id"]
                )
            else:
                stmt = sqlite_insert(NewsSentiment).values(articles).on_conflict_do_nothing(
                    index_elements=["entry_id"]
                )
            result = await self._session.execute(stmt)
            await self._session.commit()
            return result.rowcount or 0

        async def save_one(self, article: dict) -> Optional[NewsSentiment]:
            """Insert single article; return row if inserted, None if duplicate."""
            row = NewsSentiment(**article)
            self._session.add(row)
            try:
                await self._session.commit()
                await self._session.refresh(row)
                return row
            except IntegrityError:
                await self._session.rollback()
                return None

        async def get_records(
            self,
            since: datetime,
            until: Optional[datetime] = None,
        ) -> list[NewsSentiment]:
            if until is None:
                until = datetime.now(timezone.utc)
            stmt = (
                select(NewsSentiment)
                .where(NewsSentiment.published_at >= since)
                .where(NewsSentiment.published_at <= until)
                .order_by(NewsSentiment.published_at.asc())
            )
            result = await self._session.execute(stmt)
            return list(result.scalars().all())

        async def prune_older_than(self, days: int) -> int:
            cutoff = datetime.now(timezone.utc) - timedelta(days=days)
            stmt = delete(NewsSentiment).where(NewsSentiment.published_at < cutoff)
            result = await self._session.execute(stmt)
            await self._session.commit()
            return result.rowcount or 0
    ```
